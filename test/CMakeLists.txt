list(APPEND tests
  fill
  get
  index-of
  index-of-indexed
  index-update
  index-update-sweep

  target_link_libraries(${test} quickbit_static)

  add_test(

    COMMAND ${test}
    WORKING_DIRECTORY ${PROJECT_SOURCE_DIR}
  )
endforeach()
